017  (defn take-until
018    "Takes as parameters one number x and list of elements.
019    Returns an array, where first element is x first element of
020    given array and the second element is an array including the rest."
021    [how-many elems]
022    (loop [x how-many
023           elems elems
024           result []
025           rest-elems elems]
026      (if (zero? x)
027        [result (vec rest-elems)]
028        (recur
029         (dec x)
030         (rest elems)
031         (conj result (first elems))
032         (rest rest-elems)))))

034  (defn make-matrix
035    "Takes as a parameters the number of rows x, the number of columns y
036    and all elements of matrix in an array.
037    Composes a presentation of matrix as an array with x elements which
038    presents the rows. Each row contains y elements."
039    [rows columns elems]
040    (if (not= (* rows columns)(count elems))
041      (throw (Exception.
042              (str "You should give: amount of rows, amount of columns and array of "
043              (* rows columns) " elements.")))
044      (loop [i rows
045             elems elems
046             matrix []]
047        (if (zero? i)
048          matrix
049          (let [row-tookd (take-until columns elems)
050                row (first row-tookd)
051                rest-elems (second row-tookd)]
052            (recur (dec i) rest-elems (conj matrix row)))))))
