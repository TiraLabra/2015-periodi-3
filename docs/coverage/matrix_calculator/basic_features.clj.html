004  (defn each-elem-in-matrices
005    [matrix-a matrix-b function]
006      (if
007        (or
008         (not= (count matrix-a) (count matrix-b))
009         (not= (count (first matrix-a)) (count (first matrix-b))))
010          (matrices-are-not-valid-type)
011          (let [add-row (fn [row-a row-b] (vec (map function row-a row-b)))]
012             (vec (map add-row matrix-a matrix-b)))))

015    "Takes two matrices as parameters. If the matrices are same type, sums them:
016    To say, add every element of matrix-a to an according one of b. Returns an new matrix of same type.
017    (A + B)(i,j) = A(i, j) + B(i, j) for all i to m and j to n, where m*n is the size/type of the matrices.
018    http://en.wikipedia.org/wiki/Matrix_addition"

029  (defn scalar-product
030    "Takes one number and one matrix as parameters. Multiples every element of matrix by the scalar.
031    Returns a new matrix, which applies to formula (cA)(i,j) = c * A(i, j) where c is the scalar,
032    A a matrix and i and j the indexes of matrix."
033    [scalar matrix]
034    (if (or (not (is-matrix? matrix)) (not (number? scalar)))
035      (throw (Exception. "You should give one number and one valid matrix."))
036      (make-matrix (count matrix) (count (first matrix))
037                   (map (fn [elem] (* elem scalar))
038                        (apply concat matrix)))))
